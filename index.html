-- 超省控线分差查询（核心业务逻辑）
SELECT m.major_name, f.min_score - s.control_line AS score_diff
FROM fact_enrollment f
JOIN dim_major m ON f.major_id = m.major_id
JOIN dim_score_line s ON f.score_id = s.score_id
WHERE f.year = 2024 AND f.province_id = 1
ORDER BY score_diff DESC;
